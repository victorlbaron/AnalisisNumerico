•	library(bezier) -> Toolkit for Bezier Curves and Splines
•	library(plot3D) -> Plotting Multi-Dimensional Data
•	library(rgl) -> 3D visualization Using OpenGL
•	library(scatterplot3d) -> 3D Scatter plot
